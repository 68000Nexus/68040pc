\usepackage{bytefield}


\newcommand{\memsection}[4]{
	\bytefieldsetup{bitheight=#3\baselineskip}
	\bitbox[]{10}{
		\texttt{#1}
		\\ \vspace{#3\baselineskip} \vspace{-2\baselineskip} \vspace{- #3pt}
		\texttt{#2}
	}
	\bitbox{16}{#4}
}
